```python
class _SimWorker(QObject):
    finished = Signal(object)
    error = Signal(str)
    progress = Signal(int, str)  # percent (0-100), message string

    def __init__(self, project, mode):
        super().__init__()
        self._project = project
        self._mode = mode
        self._cancel_requested = False

    def request_cancel(self):
        self._cancel_requested = True

    def run(self):
        try:
            if self._mode == "steady":
                self.progress.emit(0, "Solving steady-state...")
                result = SteadyStateSolver().solve(self._project)
                self.progress.emit(100, "Complete")
            else:
                result = TransientSolver().solve(
                    self._project,
                    on_progress=self._on_progress,
                    cancel_check=lambda: self._cancel_requested,
                )
                if self._cancel_requested:
                    self.progress.emit(0, "Cancelled")
                else:
                    self.progress.emit(100, "Complete")
            self.finished.emit(result)
        except Exception as exc:
            self.error.emit(str(exc))

    def _on_progress(self, step, n_steps, t_max_c):
        pct = int(100 * step / max(n_steps, 1))
        self.progress.emit(pct, f"Step {step}/{n_steps} | T_max: {t_max_c:.1f} C")
```

Create `SimulationController(QObject)`:

```python
class SimulationController(QObject):
    progress_updated = Signal(int, str)  # percent, message
    run_finished = Signal(object)        # result object
    run_error = Signal(str)              # error message
    run_started = Signal()               # emitted when run begins
    run_ended = Signal()                 # emitted when run ends (success, error, or cancel)

    def __init__(self, parent=None):
        super().__init__(parent)
        self._thread = None
        self._worker = None

    @property
    def is_running(self) -> bool:
        return self._thread is not None and self._thread.isRunning()

    def start_run(self, project, mode):
        if self.is_running:
            return
        self._worker = _SimWorker(project, mode)
        self._thread = QThread()
        self._worker.moveToThread(self._thread)
        # Wire signals
        self._thread.started.connect(self._worker.run)
        self._worker.progress.connect(self.progress_updated)
        self._worker.finished.connect(self._on_finished)
        self._worker.error.connect(self._on_error)
        self._worker.finished.connect(self._thread.quit)
        self._worker.error.connect(self._thread.quit)
        self._thread.finished.connect(self._cleanup)
        self.run_started.emit()
        self._thread.start()

    def cancel(self):
        if self._worker:
            self._worker.request_cancel()

    def _on_finished(self, result):
        self.run_finished.emit(result)
        self.run_ended.emit()

    def _on_error(self, message):
        self.run_error.emit(message)
        self.run_ended.emit()

    def _cleanup(self):
        if self._worker:
            self._worker.deleteLater()
            self._worker = None
        if self._thread:
            self._thread.deleteLater()
            self._thread = None
```
