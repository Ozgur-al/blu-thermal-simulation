1. **Add temperature overlay method to Assembly3DWidget**: Create `update_temperature(self, project, result)`:
   ```python
   def update_temperature(self, project: DisplayProject, result) -> None:
       """Overlay temperature data on the assembly geometry."""
       self._plotter.clear()

       # Get global temperature range for consistent colormap
       all_temps = result.temperatures_c
       t_min = float(np.min(all_temps))
       t_max = float(np.max(all_temps))
       if t_max - t_min < 0.01:
           t_max = t_min + 1.0  # avoid zero-range colormap

       z_base = 0.0
       explode = self._explode_slider.value() / 100.0
       max_gap = sum(l.thickness for l in project.layers) * 0.5 * 1000  # mm

       nz_per_layer = getattr(result, 'nz_per_layer', [1] * len(project.layers))
       z_offsets = getattr(result, 'z_offsets', list(range(len(project.layers) + 1)))

       for l_idx, layer in enumerate(project.layers):
           t_mm = layer.thickness * 1000
           z_offset = z_base + l_idx * explode * max_gap

           # Use the top sublayer's temperature for display
           z_flat = z_offsets[l_idx] + nz_per_layer[l_idx] - 1
           temp_map = result.temperatures_c[z_flat]  # (ny, nx)
           ny, nx = temp_map.shape

           # Create structured grid for this layer
           dx = project.width * 1000 / nx
           dy = project.height * 1000 / ny
           grid = pv.ImageData(
               dimensions=(nx + 1, ny + 1, 2),
               spacing=(dx, dy, t_mm),
               origin=(0, 0, z_offset),
           )
           grid.cell_data["Temperature [C]"] = temp_map.ravel()

           self._plotter.add_mesh(
               grid,
               scalars="Temperature [C]",
               cmap="hot",
               clim=(t_min, t_max),
               show_scalar_bar=(l_idx == 0),  # one colorbar
               scalar_bar_args={"title": "Temperature [C]"},
           )

           z_base += t_mm

       self._plotter.reset_camera()
       self._plotter.render()
   ```
